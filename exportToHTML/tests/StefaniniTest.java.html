    @Test
    public void errorName() {
        navegador.get("http://prova.stefanini-jgr.com.br/teste/qa/");

        navegador.findElement(By.id("email")).sendKeys("rickvilhenaaa@gmail.com");
        navegador.findElement(By.id("password")).sendKeys("12345678");
        navegador.findElement(By.id("register")).click();

        WebElement mensagemError = navegador.findElement(By.cssSelector("p[class='error']"));
        String mensagem = mensagemError.getText();
        assertEquals("O campo Nome é obrigatório.", mensagem);
    }
